import io;
class Fac {
    int a;
    int[] b;
    public int compFac(int num){
        int num_aux;
        if (num < a)
            a = b[0];
        else
            num_aux = num * (this.compFac(num-1));
        return num_aux;
    }

    public static void main(String[]args){
        io.println(new Fac().compFac(10));
    }
}
