import io;
class Fac {
    int[] a;
    public int compFac(int[] num){
        int num_aux;
        int i;
        if (num[i] < a[0])
            i = a[i];
        else
            num_aux = num[i];

        return num_aux;
    }

    public static void main(String[]args){
        io.println(new Fac().compFac(10));
    }
}
