import io;
class Fac {

    public int compFac(int num){
        int[] a;
        int num_aux;
        int i;

        if (num < a.length)
            i = a.length;
        else
            num_aux = num;

        i = num_aux + this.compFac(num_aux - 1);
        return num_aux;
    }

    public static void main(String[]args){
        io.println(new Fac().compFac(10));
    }
}
