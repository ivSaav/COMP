import io;
class Fac {
    int a;
    public int compFac(int[] num){
        int num_aux;
        int[] b;
        int i;
        if (num[i] < a)
            a = b[i];
        else
            num_aux = num[i];
        a = a + b[i] + 1;
        return num_aux;
    }

    public static void main(String[]args){
        io.println(new Fac().compFac(10));
    }
}
